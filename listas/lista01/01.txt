para i = 1 até n - 1 faça 
  
  auxiliar = v[i]
  j = (i-1)

  enquanto ( j >= 0 ) e ( auxiliar < v[j] ) 
    v[j+1] = v[j]
    j--
  fim_enquanto
  
  se (j!=(i-1)) então
    v[j+1] = auxiliar
  fim_se
